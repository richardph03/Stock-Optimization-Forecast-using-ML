19def get_supabase_client() -> Client | None: 

20 """

23 Returns:

36def save_results_to_supabase(result: dict[str, Any]) -> None: 

37 """ 

38 Save optimisation results to Supabase database. 

39 

40 Args: 

41 result: Dictionary containing optimisation results from run_optimisation() 

42 Expected keys: predictions, predicted_returns, weights 

43 

44 Raises: 

45 ValueError: If Supabase client cannot be created or insertion fails 

46 """ 

47 supabase = get_supabase_client() 

48 if supabase is None: 

49 raise ValueError( 

50 "Supabase client not available. Check SUPABASE_URL and SUPABASE_KEY environment variables." 

51 ) 

52 

53 as_of_date = result.get("date") 

54 predictions = result.get("predictions", {}) 

55 predicted_returns = result.get("predicted_returns", {}) 

56 weights = result.get("weights", {}) 

57 actual_prices_last_month = result.get("actual_prices_last_month", {}) 

58 

59 if not predictions: 

60 logger.warning("No predictions to save") 

61 return 

62 

63 # Prepare rows for insertion - one row per stock 

64 rows = [] 

65 for ticker in predictions.keys(): 

66 row = { 

67 "id": str(uuid.uuid4()), 

68 "created_at": datetime.now().isoformat(), 

69 "as_of_date": as_of_date.isoformat() if as_of_date else None, 

70 "ticker": ticker, 

71 "predicted_price": float(predictions.get(ticker, 0.0)), 

72 "predicted_return": float(predicted_returns.get(ticker, 0.0)), 

73 "actual_prices_last_month": json.dumps(actual_prices_last_month.get(ticker, [])), 

74 "portfolio_weight": float(weights.get(ticker, 0.0)), 

75 } 

76 rows.append(row) 

77 

78 logger.info(f"Inserting {len(rows)} rows into Supabase...") 

79 (supabase.table(SUPABASE_TABLE_NAME).insert(rows).execute()) 

80 

81 logger.info(f"Successfully saved {len(rows)} predictions to Supabase")
